class Abbrlink
{
private:
    int abbr[8];

public:
    void New()
    {
        int seed = (int)time(0);
        for (int i = 0; i < 8; i++)
        {
            int bytes = -1;
            while ((bytes < 0) || (bytes > 15))
            {
                seed--;
                srand(seed);
                bytes = rand() % 16;
            }
            abbr[i] = bytes;
        }
    }

    void Print()
    {
        for (int i = 0; i < 8; i++)
        {
            cout << hex << abbr[i];
        }
        cout << endl;
    }
};
